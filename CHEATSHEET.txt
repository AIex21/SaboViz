Save images
docker save -o sabo-viz.tar postgres:15 sabo-rascal-parser:latest visualizationtool-backend:latest visualizationtool-frontend:latest

Load images
podman load -i sabo-viz.tar

Run prod
podman-compose -f podman-compose.prod.yml up

See all containers
podman ps -a

Delete containers
podman container prune  ||  podman rm -af

View the container's logs
podman logs -f sabo_backend

See all images
podman images

Delete images
podman image prune -a   ||  podman rmi -a -f

See all volumes
podman volume ls

Delete volumes
podman volume prune     ||  podman volume rm -a

Stopping and Deleting everything
podman stop -a
podman system prune -a --volumes -f
